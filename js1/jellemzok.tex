\begin{frame}
    A JavaScript fontosabb tulajdonságai
    \begin{itemize}
      \item Web oldalba ágyazható, de \hiv{\href{https://nodejs.org/en/about/}{szerver (backend) oldalon}} is használható
      \item Objektum-orientált, de \kiemel{nincsenek osztályok} (objektum alapú,
  vagy prototípus alapú)
      \item Gyengén típusos, dinamikusan típusos (változók futásidőben típust
  válthatnak) $\rightarrow$ kényelmes, de veszélyes
      \item Böngésző által értelmezett (interpreter/JIT) nyelv (script)
      \item HTML DOM (Document Object Model) elérése
      \item Eseménykezelés
      \item Dokumentáció:
  \hiv{\href{https://www.ecma-international.org/publications-and-standards/standards/ecma-262/
  }{ECMA}}, 
  \hiv{\href{https://developer.mozilla.org/en-US/docs/Web/JavaScript}{Mozilla}}
    \end{itemize}
  \end{frame}
